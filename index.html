What is easy.js?

easy js is a tool to make your javascript less better and easy
with easy js you can do stuff like model box using one line other than 70!!!


Let us now look a example program :

import * as easy from "easy.js"
easy.model("hello world", "i made this model by <br>using easy.js", "css width", "css heigth", "close", "background color", "text color")
//you can use null to put the defalt value
//you can use any heigth and width and close text
//you can use any background and text defalt is text=black background=white

getting elements using easy.js

using easy.js you can get elements by DOM much easier
easy.js give ten options alternatives for DOM

ele()

use ele("#id") to get an element by id
example import * as easy from "easy.js"
easy.ele("#spam")
//try this in console use ele(".id") to get an element by id(first element with that class)
example import * as easy from "easy.js"
easy.ele(".spam")
//try this in console

ele_id()

use ele_id("id") to get an element by id
example import * as easy from "easy.js"
easy.ele_id("eggs")
//try this in console

eles_class()

use ele_class("classname") to get all elements using a classname
example import * as easy from "easy.js"
easy.eles_class("eggs")
//try this in console

eles_tagname()

use eles_tagname(name) to get all elements using that tag name
example import * as easy from "easy.js"
easy.eles_tagname("code")
//try this in console

intext()

use intext(".classname"/"#id") to get the inner text of a element with a id/classname
example import * as easy from "easy.js"
easy.intext("#foo")
//try this in console

intext_s()

use intext_s(".classname"/"#id" , "text to replace") to set the inner text of a element with a id/classname
example import * as easy from "easy.js"
easy.intext_s(".testy", "text changed")
//try this in console

inhtml()

use inhtml(".classname"/"#id") to get the inner html of a element with a id/classname
example import * as easy from "easy.js"
easy.inhtml("#foo")
//try this in console

inhtml_s()

use inhtml_s(".classname"/"#id", "changed") to change the inner html of a element with a id/classname
example import * as easy from "easy.js"
easy.inhtml_s(".la", "cool?")
//try this in console

src()

use src(".classname"/"#id") to get the src arrtibute of a element with a id/classname
example import * as easy from "easy.js"
easy.src("#script")
//try this in console

src_s()

use src_s(".classname"/"#id") to set the src arrtibute of a element with a id/classname
example import * as easy from "easy.js"
easy.src_s(".bar", "./easy.js")
//try this in console

localstorage

easy.js give alternative of localstorage comands

ls()

this will retrun the localstorage ls() import * as easy from "easy.js"
easy.ls()

ls_s()

this is used to set somthing in localstorage use ls_s(id, value) this will also retrun a item id set import * as easy from "easy.js"
easy.ls_s(id, value)

ls_g()

this is used to get somthing from localstorage use ls_g(id) this will retrun the item import * as easy from "easy.js"
easy.ls_g(id)

ls_c()

this is used to clear the localstorage ls_c() import * as easy from "easy.js"
easy.ls_c()

ls_r()

this is used to remove a item from localstorage with that id import * as easy from "easy.js"
easy.ls_c(id)
